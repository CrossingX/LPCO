main:

	li	$t1, 56		#Y

	slti 	$t2 $t0, 50 	#ÅÐ¶ÏKÊÇ·ñÐ¡ÓÚ50

	sll	$t5, $t4, 4	#$t5 = 16*(K/4+210)
